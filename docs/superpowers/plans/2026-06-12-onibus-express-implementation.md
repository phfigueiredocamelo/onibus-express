```csharp
namespace OnibusExpress.Domain.Services;

public static class CpfValidator
{
    public static bool IsValid(string? cpf)
    {
        var digits = new string((cpf ?? string.Empty).Where(char.IsDigit).ToArray());
        if (digits.Length != 11 || digits.Distinct().Count() == 1)
            return false;

        var numbers = digits.Select(c => c - '0').ToArray();
        var first = CalculateDigit(numbers, 9, 10);
        var second = CalculateDigit(numbers, 10, 11);

        return numbers[9] == first && numbers[10] == second;
    }

    private static int CalculateDigit(int[] numbers, int length, int weight)
    {
        var sum = 0;
        for (var i = 0; i < length; i++)
            sum += numbers[i] * (weight - i);

        var remainder = sum % 11;
        return remainder < 2 ? 0 : 11 - remainder;
    }
}
```

```csharp
// backend/src/OnibusExpress.Application/Reservations/CreateReservationService.cs
using OnibusExpress.Application.Abstractions;
using OnibusExpress.Application.Common;
using OnibusExpress.Domain.Entities;
using OnibusExpress.Domain.Enums;
using OnibusExpress.Domain.Services;

namespace OnibusExpress.Application.Reservations;

public class CreateReservationService(IReservationRepository repository, IReservationCodeGenerator codeGenerator)
{
    public async Task<Result<ReservationDto>> CreateAsync(CreateReservationRequest request, DateTimeOffset now, CancellationToken cancellationToken = default)
    {
        if (!CpfValidator.IsValid(request.PassengerCpf))
            return Result<ReservationDto>.Fail("CPF inválido.");

        var trip = await repository.GetTripWithReservationsAsync(request.TripId, cancellationToken);
        if (trip is null)
            return Result<ReservationDto>.Fail("Viagem não encontrada.");

        if (trip.HasDeparted(now))
            return Result<ReservationDto>.Fail("Não é possível reservar uma viagem já realizada.");

        if (request.SeatNumber < 1 || request.SeatNumber > trip.TotalSeats)
            return Result<ReservationDto>.Fail("Assento inválido.");

        var occupied = trip.Reservations.Any(r => r.SeatNumber == request.SeatNumber && r.Status == ReservationStatus.Active);
        if (occupied)
            return Result<ReservationDto>.Fail("Assento já ocupado.");

        var reservation = new Reservation
        {
            Id = Guid.NewGuid(),
            TripId = trip.Id,
            Trip = trip,
            SeatNumber = request.SeatNumber,
            Status = ReservationStatus.Active,
            CreatedAt = now,
            Code = await codeGenerator.GenerateAsync(cancellationToken),
            Passenger = new Passenger
            {
                Id = Guid.NewGuid(),
                FullName = request.PassengerName.Trim(),
                Cpf = request.PassengerCpf,
                Email = request.PassengerEmail.Trim()
            }
        };

        await repository.AddAsync(reservation, cancellationToken);
        await repository.SaveChangesAsync(cancellationToken);

        return Result<ReservationDto>.Ok(ToDto(reservation));
    }

    public static ReservationDto ToDto(Reservation reservation) => new(
        reservation.Code,
        reservation.Trip.Id,
        reservation.Trip.Route.Origin,
        reservation.Trip.Route.Destination,
        reservation.Trip.DepartureAt,
        reservation.Trip.BasePrice,
        reservation.SeatNumber,
        reservation.Passenger.FullName,
        reservation.Passenger.Cpf,
        reservation.Passenger.Email,
        reservation.Status.ToString());
}
```

```csharp
// backend/src/OnibusExpress.Application/Reservations/CancelReservationService.cs
using OnibusExpress.Application.Abstractions;
using OnibusExpress.Application.Common;
using OnibusExpress.Domain.Enums;

namespace OnibusExpress.Application.Reservations;

public class CancelReservationService(IReservationRepository repository)
{
    public async Task<Result> CancelAsync(string code, DateTimeOffset now, CancellationToken cancellationToken = default)
    {
        var reservation = await repository.GetByCodeAsync(code, cancellationToken);
        if (reservation is null)
            return Result.Fail("Reserva não encontrada.");

        if (reservation.Status == ReservationStatus.Cancelled)
            return Result.Fail("Reserva já cancelada.");

        if (!reservation.Trip.CanCancel(now))
            return Result.Fail("Cancelamento permitido apenas até 2 horas antes da partida.");

        reservation.Status = ReservationStatus.Cancelled;
        reservation.CancelledAt = now;
        await repository.SaveChangesAsync(cancellationToken);

        return Result.Ok();
    }
}
```

```csharp
using Microsoft.EntityFrameworkCore;
using OnibusExpress.Domain.Entities;
using OnibusExpress.Domain.Enums;

namespace OnibusExpress.Infrastructure.Data;

public class OnibusExpressDbContext(DbContextOptions<OnibusExpressDbContext> options) : DbContext(options)
{
    public DbSet<BusRoute> Routes => Set<BusRoute>();
    public DbSet<Trip> Trips => Set<Trip>();
    public DbSet<Passenger> Passengers => Set<Passenger>();
    public DbSet<Reservation> Reservations => Set<Reservation>();

    protected override void OnModelCreating(ModelBuilder modelBuilder)
    {
        modelBuilder.Entity<BusRoute>().ToTable("Routes");
        modelBuilder.Entity<BusRoute>().Property(r => r.Origin).HasMaxLength(120).IsRequired();
        modelBuilder.Entity<BusRoute>().Property(r => r.Destination).HasMaxLength(120).IsRequired();

        modelBuilder.Entity<Trip>().Property(t => t.BasePrice).HasPrecision(10, 2);
        modelBuilder.Entity<Trip>().HasOne(t => t.Route).WithMany(r => r.Trips).HasForeignKey(t => t.RouteId);
```

```csharp
public class ReservationCodeGenerator(IReservationRepository repository) : IReservationCodeGenerator
{
    public async Task<string> GenerateAsync(CancellationToken cancellationToken = default)
    {
        for (var attempt = 0; attempt < 20; attempt++)
        {
            var number = Random.Shared.Next(10000, 99999);
            var code = $"ONB-{number}";
            if (!await repository.CodeExistsAsync(code, cancellationToken))
                return code;
        }

        return $"ONB-{Guid.NewGuid():N}"[..13].ToUpperInvariant();
    }
}
```

```ts
export function isValidCpf(value: string): boolean {
  const digits = value.replace(/\D/g, "");
  if (digits.length !== 11 || /^(\d)\1+$/.test(digits)) return false;
  const numbers = digits.split("").map(Number);
  const calc = (length: number) => {
    const sum = numbers.slice(0, length).reduce((acc, digit, index) => acc + digit * (length + 1 - index), 0);
    const remainder = sum % 11;
    return remainder < 2 ? 0 : 11 - remainder;
  };
  return numbers[9] === calc(9) && numbers[10] === calc(10);
}
```
